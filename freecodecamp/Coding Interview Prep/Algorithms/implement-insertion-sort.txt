function insertionSort(array) {
  // change code below this line

  let k = 0;
  let j = 0;
  for(let i=1; i<array.length; ++i) {
    k = array[i];
    j = i;
    while(j > 0 && k < array[j-1]) {
      let x = array[j-1];
      array[j-1] = k;
      array[j] = x;
      --j;
    }
  }

  return array;
  // change code above this line
}

insertionSort([1, 4, 2, 8, 345, 123, 43, 32, 5643, 63, 123, 43, 2, 55, 1, 234, 92]);
