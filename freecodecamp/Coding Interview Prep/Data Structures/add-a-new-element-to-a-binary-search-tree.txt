function Node(value) {
  this.value = value;
  this.left = null;
  this.right = null;
}
function BinarySearchTree() {
  this.root = null;
  // change code below this line
  this.isLeafNode = function(node) {
    return (this.left === null && this.right === null);
  }

  this.add = function(value) {
    let newNode = new Node(value);

    if(this.root === null) {
      this.root = newNode;
      return;
    }

    let cur = this.root;
    while(true) {
      if(this.isLeafNode(cur)) {
        if(cur.value > value) {
          cur.left = newNode
        }
        else if(cur.value < value) {
          cur.right = newNode;
        }
        else return null;

        return;
      }
      if(cur.value > value) {
        if(cur.left !== null)
          cur = cur.left;
        else {
          cur.left = newNode;
          return;
        }
      }
      else if(cur.value < value) {
        if(cur.right !== null)
          cur = cur.right;
        else {
          cur.right = newNode;
          return;
        }
      }
      else {
        return null;
      }
    }

  }

}
