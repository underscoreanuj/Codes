function Node(value) {
  this.value = value;
  this.left = null;
  this.right = null;
}
function BinarySearchTree() {
  this.root = null;
  // change code below this line
  this.findMin = function() {
    if(this.root === null)
      return null;

    let cur = this.root;
    while(cur.left !== null)
      cur = cur.left;

    return cur.value;
  }

  this.findMax = function() {
    if(this.root === null)
      return null;

    let cur = this.root;
    while(cur.right !== null)
      cur = cur.right;

    return cur.value;
  }
  // change code above this line
}
